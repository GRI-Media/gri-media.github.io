    <script src="https://cdn.jsdelivr.net/npm/jquery@3.6.0/dist/jquery.min.js"></script>

    <script src="https://cdn.jsdelivr.net/npm/spritespin@4.0.11/release/spritespin.min.js"></script>

  </head>

  <body>

    <h1>Aegirine Viewer</h1>

    <div id="aegirine-viewer"></div>

    <button id="toggle-spin">Pause Rotation</button>


    <script>

      const imageUrls = [];

      for (let i = 1; i <= 80; i++) {

        const num = i.toString().padStart(5, "0");

        imageUrls.push(`https://res.cloudinary.com/dy8hjxyum/image/upload/Aegirine-${num}.jpg`);

      }


      let isPlaying = true;


      $("#aegirine-viewer").spritespin({

        source: imageUrls,

        width: 500,

        height: 500,

        sense: -1,

        frameTime: 60,

        animate: isPlaying,

        loop: true,

        responsive: true,

      });


      $("#toggle-spin").click(function () {

        isPlaying = !isPlaying;

        $("#aegirine-viewer").spritespin("animate", isPlaying);

        $(this).text(isPlaying ? "Pause Rotation" : "Resume Rotation");

      });

    </script>

  </body>
